DATA A
DATA B
DATA C
DATA D
START:
READ AX
MOV A,AX
READ BX
MOV B,BX
READ CX
MOV C,CX
IF AX GT BX THEN
    IF AX GT CX THEN
        MOV DX,A
    ENDIF
ENDIF
IF BX GT AX THEN
    IF BX GT CX THEN
        MOV DX,B
    ENDIF
ENDIF
IF CX GT BX THEN
    IF CX GT AX THEN
        MOV DX,C
    ENDIF
ENDIF
MOV D,DX
PRINT D
END
